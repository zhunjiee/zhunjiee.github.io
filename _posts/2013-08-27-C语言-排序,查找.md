#排序
###1.选择排序
```c
int main(int argc, const char * argv[]) {
    /*
     选择排序:
     第一轮拿数组的第一个数依次和后面的数进行比较,一轮过后比较出最小的数放在数组第一位
     第二轮从数组的第二个数依次和后面的数进行比较......，比较过后第二小的数出现在数组的第二位
     。。。。。。
     类似打印倒三角，需修改内层for循环第一个

     */

    int arr[] = {254, 2, 8567, 254, 6785, 68, 5};
    int length = sizeof(arr) / sizeof(arr[0]);

    for (int i = 0; i < length - 1; i++) {
        for (int j = i + 1; j < length; j++) {
            if (arr[i] > arr[j]) {
                int temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
        }
    }

    for (int i = 0; i < length; i++) {
        printf("arr[%d] = %d\n", i + 1, arr[i]);
    }

    return 0;
}

```
###2.冒泡排序
```c
int main(int argc, const char * argv[]) {
    /*
     冒泡排序：
     第一个数先和相邻的第二个数比较，较大的数出现在第二位，再用第二位的数和相邻第三位的数比较，较大的数出现在第三位，依次类推，比较length-1次（因为最后一个不用继续比较了），一轮过后最大的数出现在了数组最后面；
     第二轮继续用第一个数和相邻的第二个数比较，。。。。。。比较（length-1）- 1 次之后，第二大的数出现在数组倒数第二位

     类似打印正三角，修改内层循环的第二个数

     */

    int arr[] = {254, 2, 8567, 254, 6785, 68, 5};
    int length = sizeof(arr) / sizeof(arr[0]);

    for (int i =0; i < length - 1; i++) { // 外层控制的是比较的轮数
        for (int j = 0; j < length - 1 - i; j++) { // 内层控制的比较次数
            if (arr[j] > arr[j+1]) {
                int temp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = temp;
            }
        }
    }

    for (int i = 0; i < length; i ++) {
        printf("arr[%i] = %i\n", i, arr[i]);
    }


    return 0;
}
```

```c
// 只要还在范围内就需要查找
int findKey3(int arr[], int length, int key){
    int min = 0;
    int max = length - 1;

    // 当min小于max时，一直查找
    while(min <= max) {
        // 计算中间值
        int mid = (min + max) / 2;
        if (key > arr[mid]){
            min = mid + 1;
        }else if (key < arr[mid]){
            max = mid - 1;
        }else{
            return mid;
        }
    }
    return -1;
}

int findKey2(int arr[], int length, int key){
    int min = 0;
    int max = length - 1;
    int mid = (min + max) / 2;

    // 当我们要找的key不等于中间值时，一直查找
    while(key != arr[mid]) {
        if (key > arr[mid]){
            min = mid + 1;
        }else if (key < arr[mid]){
            max = mid - 1;
        }

        // 超出范围,返回-1
        if (min > max){
            return -1;
        }
        // 重新计算中间值
        mid = (min + max) / 2;
    }
    return mid;
}

int findKey(int arr[], int length, int key){
    for (int i = 0; i < length; i++) {
        if (arr[i] == key) {
            return i;
        }
    }
    return -1;
}
```

```c
    int min, max, mid;

int findKey2(int nums[], int length, int key)
{
    int min, max, mid;
    min = 0;
    max = length - 1;
    mid = (min + max) / 2;

    while (key != nums[mid]) {
        // 判断如果要找的值, 大于了取出的值, 那么min要改变
        if (key > nums[mid]) {
            min = mid + 1;
        // 判断如果要找的值, 小雨了取出的值, 那么max要改变
        }else if (key < nums[mid])
        {
            max = mid - 1;
        }

        // 超出范围, 数组中没有需要查找的值
        if (min > max) {
            return -1;
        }
        // 每次改变完min和max都需要重新计算mid
        mid = (min + max) / 2;
    }
//    printf("aaaaaa\n");

    return mid;

}
```
